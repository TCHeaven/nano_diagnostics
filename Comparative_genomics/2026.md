    return str(seq).upper().strip("N")

for rec in SeqIO.parse(NCBI_FASTA, "fasta"):

    ncbi_seq_to_id[seq] = rec.id

    ncbi_id_to_seq[rec.id] = seq

#original

used_ncbi = set()

    out.write("original_id\tncbi_id\n")
    for orig_id, seq in orig_id_to_seq.items():
        ncbi_id = ncbi_seq_to_id.get(seq, "NOT_FOUND")
        if ncbi_id != "NOT_FOUND":
            used_ncbi.add(ncbi_id)
        out.write(f"{orig_id}\t{ncbi_id}\n")
    for ncbi_id in ncbi_id_to_seq.keys():
        if ncbi_id not in used_ncbi:
            out.write(f"NOT_FOUND\t{ncbi_id}\n")

orig_genes = []  # (contig, gene_id)

    for line in f:
        if line.startswith("#"):
            continue
        cols = line.strip().split("\t")
        if len(cols) < 9:
            continue
        contig = cols[0]
        feature = cols[2]
        attrs = cols[8]
        if feature == "gene":
            gene_id = None
            for part in attrs.split(";"):
                if part.startswith("ID="):
                    gene_id = part.replace("ID=", "")
            if gene_id:

        else:
            ncbi_gene_list = ["NOT_FOUND"]
        for ncbi_gene in ncbi_gene_list:
            out.write(
                f"{gene_id}\t{contig}\t{ncbi_contig}\t{ncbi_gene}\n"
            )
